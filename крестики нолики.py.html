map_ = [1,2,3,
        4,5,6,
        7,8,9,]

victories = [[0,1,2],
             [3,4,5],
             [6,7,8],
             [0,3,6],
             [1,4,7],
             [2,5,8],
             [0,4,8],
             [2,4,6]]


def print_maps():
    print("----------")

    print(map_[0], end=" | ")
    print(map_[1], end=" | ")
    print(map_[2])

    print("----------")

    print(map_[3], end=" | ")
    print(map_[4], end=" | ")
    print(map_[5])

    print("----------")

    print(map_[6], end=" | ")
    print(map_[7], end=" | ")
    print(map_[8])

    print("----------")

def steps_maps(step, symbol):
    ind = map_.index(step)
    map_[ind] = symbol

def get_results():
    win = ""
    for i in victories:
        if map_[i[0]] == "X" and map_[i[1]] == "X" and map_[i[2]] == "X":
            win = "X"
        if map_[i[0]] == "O" and map_[i[1]] == "O" and map_[i[2]] == "O":
            win = "O"
    return win

game_over = False
player1 = True

while game_over == False:
    print_maps()

    if player1 == True:
        symbol = "X"
        step = int(input("Первый игрок, выбери ход: "))
    else:
        symbol = "O"
        step = int(input("Второй игрок, выбери ход: "))
    steps_maps(step, symbol)
    win = get_results()
    if win != "":
        game_over = True
    else:
        game_over = False

    player1 = not(player1)
print_maps()
print("Победитель", win)
